14      private static final byte[] buffer = new byte[BUFF_SIZE];

16      /**
17       * Copy an InputStream to an OutputStream. 
18       * While this method will automatically close the destination OutputStream,
19       * the caller is responsible for closing the source InputStream.
20       * 
21       * @param source
22       * @param destination
23       * @return <code>true</code> if the operation was successful;
24       *         <code>false</code> otherwise (which includes a null input).
25       * @see http://java.sun.com/docs/books/performance/1st_edition/html/JPIOPerformance.fm.html#22980
26       */
27      public static boolean copy(InputStream source, OutputStream destination) {
28          try {
29              while (true) {
30                  synchronized (buffer) {
31                      int amountRead = source.read(buffer);
32                      if (amountRead == -1) {
33                          break;
34                      }
35                      destination.write(buffer, 0, amountRead);
36                  }
37              }
38              destination.flush();
39              destination.close();
40              return true;
41          } catch (IOException e) {
42              return false;
43          }
44      }
