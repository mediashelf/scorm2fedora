33  /**
34   * Scorm2Fedora accepts a SCORM 1.2 package with IMS Metadata and deposits it
35   * into a Fedora repository. The IMS Metadata is converted into Dublin Core and
36   * made available the Fedora object's DC datastream.

38   * @author Edwin Shin
39   *
40   */
41  public class Scorm2Fedora {

49  	 * <dl>
50  	 *   <dt>username
51  	 *   <dd>username for the Fedora repository to be used by Scorm2Fedora to
52  	 *   deposit SCORM packages.
53  	 *   <dt>password
54  	 *   <dd>password for the Fedora user
55  	 *   <dt>baseUrl
56  	 *   <dd>URL of the Fedora repository (e.g. http://localhost:8080/fedora)
57  	 *   <dt>namespace

59  	 *   <dt>cmodel
60  	 *   <dd>The URI of the cmodel newly created objects (optional)

67  	public Scorm2Fedora(Properties props) {
68  		if (props == null) {
69  			logger.debug("null properties file, will use defaults");
70  			props = new Properties();
71  			try {
72  				props.load(getClass().getResourceAsStream("/scorm2fedora.properties"));
73  			} catch (IOException e) {
74  				logger.warn("unable to load properties from classpath");
75  			}
76  		}
77  		this.props = props;
78  
79  		String username = props.getProperty("username", "fedoraAdmin");
80  		String password = props.getProperty("password", "fedoraAdmin");
81  		URL baseUrl;
82  		try {
83  			baseUrl = new URL(props.getProperty("baseUrl",
84  					"http://localhost:8080/fedora"));
85  		} catch (MalformedURLException e) {
86  			throw new IllegalArgumentException("Malformed baseUrl: "
87  					+ e.getMessage(), e);
88  		}
89  		fedora = new FedoraClient(new FedoraCredentials(baseUrl, username,
90  				password));
91  	}

93  	/**
94  	 *
95  	 * @param in the SCORM package
96  	 * @param filename filename of the SCORM package, used as the dsLabel
97  	 * @return
98  	 * @throws IOException
99  	 * @throws CrosswalkException
100 	 * @throws FedoraClientException
101 	 */
102 	public ScormDeposit deposit(InputStream in, String filename) throws IOException, CrosswalkException, FedoraClientException {
103 		Set<File> tempFiles = new HashSet<File>();
104 
105 		String pid = null;
106 		URI location = null;
107 		try {
108 			// create a temp file for the scorm package
109 			File scorm = File.createTempFile("scorm", null);
110 			tempFiles.add(scorm);
111 			OutputStream fout = new FileOutputStream(scorm);
112 			FileUtils.copy(in, fout);
113 			in.close();
114 
115 			// extract the manifest
116 			in = new BufferedInputStream(new FileInputStream(scorm));
117 			ZipInputStream zin = new ZipInputStream(in);
118 			ZipEntry entry;
119 			File manifest = null;
120 			while ((entry = zin.getNextEntry()) != null) {
121 				if (entry.getName().equals("imsmanifest.xml")) {
122 					manifest = File.createTempFile("man", null);
123 					tempFiles.add(manifest);
124 					FileUtils.copy(zin, new FileOutputStream(manifest));
125 					break;
126 				}
127 			}
128 			zin.close();
129 			if (manifest == null) {
130 				throw new IllegalArgumentException("package has no manifest!");
131 			}
132 
133 			// convert manifest's metadata to oai_dc
134 			in = new FileInputStream(manifest);
135 			File oaidc = File.createTempFile("oaidc", null);
136 			tempFiles.add(oaidc);
137 			OutputStream out = new FileOutputStream(oaidc);
138 			imsmd2oaidc(in, out);
139 			in.close();
140 			out.close();
141 
142 			// create new fedora object
143 			String namespace = props.getProperty("namespace");
144 			IngestResponse response = ingest(pid).namespace(namespace).execute(fedora);
145 
146 			pid = response.getPid();
147 			location = response.getLocation();
148 
149 			// add original scorm package as datastream
150 			String dsId = props.getProperty("scorm.dsid", "SCORM");
151 			addDatastream(pid, dsId).controlGroup("M")
152 			                        .dsLabel(filename)
153 			                        .content(scorm).execute(fedora);
154 
155 			// update DC datastream with new oai_dc
156 			modifyDatastream(pid, "DC").content(oaidc).execute(fedora);
157 
158 			// Add content model to RELS-EXT
159 			String cmodel = props.getProperty("cmodel");
160 			if (cmodel != null) {
161     			String relsExt = new RelsExt.Builder(pid).addCModel(cmodel).build().toString();
162     			addDatastream(pid, "RELS-EXT")
163     			    .dsLabel("RDF Statements about this object")
164     			    .formatURI("info:fedora/fedora-system:FedoraRELSExt-1.0")
165     			    .mimeType("application/rdf+xml").content(relsExt)
166     			    .execute(fedora);
167     			logger.debug("Created RELS-EXT with cmodel \"{}\".", cmodel);
168 			} else {
169 			    logger.info("Property \"cmodel\" not found. Skipping RELS-EXT.");
170 			}
171 
172 		} finally {
173 			for (File f : tempFiles) {
174 				f.delete();
175 				logger.debug("Deleted temp file \"{}\".", f.getAbsolutePath());
176 			}
177 		}
178 		return new ScormDeposit(pid, location);
179 	}
