43  	    Properties props = null;
44          if (sc != null && sc.getInitParameter("baseUrl") != null) {
45              logger.debug("Using web.xml init-params for configuration");
46              props = new Properties();
47              String[] pnames = {"username", "password", "baseUrl", "namespace", "scorm.dsid", "cmodel"};
48              for (String pname : pnames) {
49                  String pval = sc.getInitParameter(pname);
50                  if (pval == null) {
51                      logger.warn("init-param {} is null", pname);
52                  } else {
53                      props.setProperty(pname, pval);
54                  }
55              }
56          } else {
57              logger.debug("configuration init-params not found. Using default properties.");
58          }
59          s2f = new Scorm2Fedora(props);
60  	}

62  	/**
63  	 * Deposits a SCORM 1.2 package with IMS Metadata into a Fedora repository.
64  	 *
65  	 * @param request the multipart/form-data request that contains the SCORM
66  	 * package
67  	 * @return the pid and location (URI) of the newly created Fedora object.
68  	 */
69  	@POST
70  	@Consumes(MediaType.MULTIPART_FORM_DATA)
71  	@Produces(MediaType.APPLICATION_JSON)
72  	public ScormDeposit deposit(@Context HttpServletRequest request) {
73  		// Check that we have a file upload request
74  		boolean isMultipart = ServletFileUpload.isMultipartContent(request);
75  		if (!isMultipart) {
76  			logger.warn("Request did not contain multipart content.");
77  			throw new WebApplicationException(Response.status(
78  					Response.Status.BAD_REQUEST).entity(
79  					"request does not contain multipart content.").type(
80  					MediaType.TEXT_PLAIN).build());
81  		}
82  
83  		// Create a new file upload handler
84  		ServletFileUpload upload = new ServletFileUpload();
85  
86  		// Parse the request
87  		try {
88  			FileItemIterator iter = upload.getItemIterator(request);
89  			while (iter.hasNext()) {
90  				FileItemStream item = iter.next();
91  				String name = item.getFieldName();
92  				InputStream stream = item.openStream();
93  				if (item.isFormField()) {
94  					logger.debug("Ignoring form field \"{}\".", name);
95  					stream.close();
96  				} else {
97  					logger.debug("Found file field {} with file name \"{}\".",
98  							name, item.getName());
99  					return s2f.deposit(stream, item.getName());
100 				}
101 			}
102 		} catch (FileUploadException e) {
103 			logger.debug(e.getMessage(), e);
104 			throw new WebApplicationException(e, Response.Status.INTERNAL_SERVER_ERROR);
105 		} catch (IOException e) {
106 			logger.debug(e.getMessage(), e);
107 			throw new WebApplicationException(e, Response.Status.INTERNAL_SERVER_ERROR);
108 		} catch (CrosswalkException e) {
109 			logger.debug(e.getMessage(), e);
110 			throw new WebApplicationException(e, Response.Status.INTERNAL_SERVER_ERROR);
111 		} catch (FedoraClientException e) {
112 			logger.debug(e.getMessage(), e);
113 			throw new WebApplicationException(e, Response.Status.INTERNAL_SERVER_ERROR);
114 		}
115 		logger.warn("Request did not contain a file.");
116 		throw new WebApplicationException(Response.status(
117 				Response.Status.BAD_REQUEST).entity(
118 				"request does not contain a file.").type("text/plain").build());
119 	}
