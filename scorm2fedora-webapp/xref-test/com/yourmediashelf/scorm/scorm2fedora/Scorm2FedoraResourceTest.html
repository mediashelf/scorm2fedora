31  public class Scorm2FedoraResourceTest extends JerseyTest {
32  
33  	public Scorm2FedoraResourceTest() {
34  		super(new WebAppDescriptor
35  			.Builder("com.yourmediashelf.scorm.scorm2fedora")
36  			.contextPath("scorm2fedora")
37  			.initParam("username", System.getProperty("fedora.test.username"))
38  			.initParam("password", System.getProperty("fedora.test.password"))
39  			.initParam("baseUrl", System.getProperty("fedora.test.baseUrl"))
40  			.initParam("namespace", "scorm-test")
41  			.build());
42  	}
43  
44  	@Test
45  	public void testDeposit() throws Exception {
46  		File pkg = new File("src/test/resources/21.edit.essay.zip");
47  		assertTrue(pkg.exists());
48  
49  		MimeUtil2 mimeUtil = new MimeUtil2();
50  		mimeUtil
51  				.registerMimeDetector("eu.medsea.mimeutil.detector.MagicMimeMimeDetector");
52  		String mimeType = MimeUtil2.getMostSpecificMimeType(
53  				mimeUtil.getMimeTypes(pkg)).toString();
54  		assertEquals("application/zip", mimeType);
55  
56  		//client().addFilter(new LoggingFilter());
57  		MultiPart multiPart = new FormDataMultiPart()
58  				.bodyPart(new FileDataBodyPart("file", pkg, MediaType
59  						.valueOf(mimeType)));
60  
61  		ScormDeposit response = resource()
62  			.type(MediaType.MULTIPART_FORM_DATA)
63  			.accept(MediaType.APPLICATION_JSON)
64  			.post(ScormDeposit.class, multiPart);
65  		// The location (e.g. http://localhost:8080/fedora/objects/scorm%3A5),
66  		// once decoded, should contain the pid (e.g. scorm:5)
67  		assertTrue(URLDecoder.decode(response.location.toString(), "UTF-8").contains(response.pid));
68  	}
