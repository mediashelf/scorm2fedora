29  public class ImsmdToOaidc implements Crosswalk {
30  	Map<String, Set<String>> supportedCrosswalks = new HashMap<String, Set<String>>();
31  	
32  	public ImsmdToOaidc() {
33  		Set<String> imsmdOutputs = new HashSet<String>();
34  		imsmdOutputs.add("oai_dc");
35  		supportedCrosswalks.put("imsmd", imsmdOutputs);
36  	}
37  	
38  	/*
39  	 * (non-Javadoc)
40  	 * 
41  	 * @see com.yourmediashelf.xwalk.Crosswalk#getSupportedCrosswalks()
42  	 */
43  	@Override
44  	public Map<String, Set<String>> getSupportedCrosswalks() {
45  		return supportedCrosswalks;
46  	}
47  
48  	/*
49  	 * (non-Javadoc)
50  	 * 
51  	 * @see com.yourmediashelf.xwalk.Crosswalk#isSupported(java.lang.String,
52  	 * java.lang.String)
53  	 */
54  	@Override
55  	public boolean isSupported(String inFormat, String outFormat) {
56  		Set<String> supported = getSupportedCrosswalks().get(inFormat);
57  		return supported != null && supported.contains(outFormat);
58  	}
59  
60  	/*
61  	 * (non-Javadoc)
62  	 * 
63  	 * @see com.yourmediashelf.xwalk.Crosswalk#translate(java.lang.String,
64  	 * java.io.InputStream, java.lang.String, java.io.OutputStream)
65  	 */
66  	@Override
67  	public void translate(String inFormat, InputStream in, String outFormat,
68  			OutputStream out) throws CrosswalkException {
69  		Source xsltSource = getXsltSource(inFormat, outFormat);
70  		TransformerFactory transFact = TransformerFactory.newInstance();
71  		transFact.setAttribute("http://saxon.sf.net/feature/version-warning",
72  				Boolean.FALSE);
73  		try {
74  			Transformer trans = transFact.newTransformer(xsltSource);
75  
76  			Source source = new StreamSource(in);
77  			StreamResult xformResult = new StreamResult(out);
78  			trans.transform(source, xformResult);
79  		} catch (TransformerException e) {
80  			throw new CrosswalkException(e.getMessage(), e);
81  		}
82  
83  	}
84  	
85  	private Source getXsltSource(String inFormat, String outFormat) {
86  		if (isSupported(inFormat, outFormat)) {
87  			return new StreamSource(getClass().getResourceAsStream("/imsmd2oaidc.xsl"));
88  		} else {
89  			throw new IllegalArgumentException("Unsupported: " + inFormat + " to " + outFormat);
90  		}
91  	}
